/**
* I am a programmer,
* so I write some code to celebrate our 1st anniversary.
*/
Boy i = new Boy("Xiaolei Liu");
Girl u = new Girl("Yiran Ding");
// Apr 23, 2013, I told you I love you.
i.love(u);
// Luckily, you accepted and became my girlfriend eversince.
u.accepted();
// Since then, I miss u every day.
i.miss(u);
// And take care of u and our love.
i.takeCareOf(u);
// You say that you won't be so easy to marry me.
// So I keep waiting and I have confidence that you will.

while (isHesitate) {
i.waitFor(u);
// I think it is an important decision
// and you should think it over.
isHesitate = u.thinkOver();
}
// After a romantic wedding, we will live happily everyday.
i.marry(u);
i.liveHappilyWith(u);
